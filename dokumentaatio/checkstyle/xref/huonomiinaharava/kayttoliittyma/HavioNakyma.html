7

9   import huonomiinaharava.huonomiinaharava.logiikka.Ruutu;
10  import huonomiinaharava.huonomiinaharava.logiikka.Ruututyyppi;
11  import javafx.scene.Scene;
12  import javafx.scene.control.Label;
13  import javafx.scene.image.Image;
14  import javafx.scene.image.ImageView;
15  import javafx.scene.layout.GridPane;
16  import javafx.scene.layout.StackPane;
17  import javafx.scene.layout.VBox;
18  import javafx.stage.Stage;
19  
20  /**
21   * Näkymä, joka kootaan pelaajan osuessa miinaan.
22   * @author anttikoivurova
23   */
24  public class HavioNakyma {
25  
26      private VBox kokonaisuus;
27      private GridPane grid;
28      private Ruudukko ruudukko;
29      private Stage stage;
30      private int aika;
31      private int vikaLeveys;
32      private int vikaKorkeus;
33      private Image miina = new Image("miina.png");
34      private Image nolla = new Image("tyhja0.png");
35      private Image ykkonen = new Image("tyhja1.png");
36      private Image kakkonen = new Image("tyhja2.png");
37      private Image kolmonen = new Image("tyhja3.png");
38      private Image nelonen = new Image("tyhja4.png");
39      private Image vitonen = new Image("tyhja5.png");
40      private Image kutonen = new Image("tyhja6.png");
41      private Image seiska = new Image("tyhja7.png");
42      private Image kasi = new Image("tyhja8.png");
43      private Image klikkaamaton = new Image("klikkaamaton.png");
44      private Image lippu = new Image("lippu.png");
45      private Image havioruutu = new Image("havioruutu.png");
46  
47      /**
48       * Luo uuden häviönäkymän.
49       * @param ruudukko pelissä käytetty ruudukko
50       * @param stage primaryStage
51       * @param aika pelin päätösaika
52       * @param vikaLeveys viimeisen klikkauksen x-koordinaatti
53       * @param vikaKorkeus viimeisen klikkauksen y-koordinaatti
54       */
55      public HavioNakyma(Ruudukko ruudukko, Stage stage, int aika, int vikaLeveys, int vikaKorkeus) {
56          this.ruudukko = ruudukko;
57          this.stage = stage;
58          this.aika = aika;
59          this.vikaLeveys = vikaLeveys;
60          this.vikaKorkeus = vikaKorkeus;
61  
62          kokoaNakyma();
63      }
64  
65      /**
66       * Kokoaa häviönäkymän.
67       */
68      public void kokoaNakyma() {
69          kokonaisuus = new VBox(10);
70          grid = new GridPane();
71          Label havioteksti = new Label(aika + "     Hävisit.");
72          kokonaisuus.getChildren().add(havioteksti);
73  
74          for (int i = 0; i < ruudukko.getKorkeus(); i++) {
75              for (int j = 0; j < ruudukko.getLeveys(); j++) {
76                  Ruutu ruutu = ruudukko.getRuudukko()[j][i];
77                  StackPane stack;
78                  if (ruutu.isKlikattu()) {
79                      if (j == vikaLeveys && i == vikaKorkeus) {
80                          stack = new StackPane(new ImageView(havioruutu));
81                      } else {
82                          stack = new StackPane(new ImageView(new Image("tyhja" + ruutu.getYmparys() + ".png")));
83                      }
84                  } else {
85                      if (ruutu.getTyyppi().equals(Ruututyyppi.MIINA)) {
86                          stack = new StackPane(new ImageView(miina));
87                      } else {
88                          stack = new StackPane(new ImageView(klikkaamaton));
89                      }
90                  }
91                  grid.add(stack, j, i);
92              }
93          }
94  
95          kokonaisuus.getChildren().add(grid);
96      }
97  
98      /**
99       * Palauttaa näkymän.
100      * @return näkymä
101      */
102     public Scene scene() {
103         return new Scene(kokonaisuus);
104     }
105 }
