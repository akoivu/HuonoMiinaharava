32      private GridPane miinaKentta;
33      private VBox kokonaisuus;
34      private Stage stage;

36      private Integer aika = 0;
37      private Label aikaLabel = new Label();
38      private int leveys;
39      private int korkeus;
40      private Timeline timeline;
41      private Image miina = new Image("miina.png");
42      private Image nolla = new Image("tyhja0.png");
43      private Image ykkonen = new Image("tyhja1.png");
44      private Image kakkonen = new Image("tyhja2.png");
45      private Image kolmonen = new Image("tyhja3.png");
46      private Image nelonen = new Image("tyhja4.png");
47      private Image vitonen = new Image("tyhja5.png");
48      private Image kutonen = new Image("tyhja6.png");
49      private Image seiska = new Image("tyhja7.png");
50      private Image kasi = new Image("tyhja8.png");
51      private Image klikkaamaton = new Image("klikkaamaton.png");
52      private Image lippu = new Image("lippu.png");
53      private Image havioruutu = new Image("havioruutu.png");

55      /**
56       * Uusi miinakenttä.
57       *
58       * @param stage primaryStage
59       * @param leveys ruudukon leveys
60       * @param korkeus ruudukon korkeus
61       * @param miinat ruudukon miinamäärä
62       */
63      public MiinaKenttaNakyma(Stage stage, int leveys, int korkeus, int miinat) {
64          this.stage = stage;
65          this.leveys = leveys;
66          this.korkeus = korkeus;
67          this.ruudukko = new Ruudukko(leveys, korkeus, miinat);
68          this.miinaKentta = new GridPane();
69          this.kokonaisuus = new VBox(10);
70          timeline = new Timeline(new KeyFrame(
71              Duration.seconds(1),
72              ae -> aika++), new KeyFrame(
73                  Duration.seconds(1),
74                  ae -> aikaLabel.setText(("" + aika.toString())))
75          );
76          timeline.setCycleCount(Animation.INDEFINITE);
77          timeline.play();
78          kokonaisuus.getChildren().add(aikaLabel);
79          kokonaisuus.getChildren().add(miinaKentta);
80          tyhjaRuudukko();
81      }
82  
83      /**
84       * Palauttaa näkymän.
85       *
86       * @return palautettava näkymä
87       */
88      public Scene scene() {
89          return new Scene(kokonaisuus);
90      }
91  
92      /**
93       * Kokoaa tyhjän miinaruudukon.
94       */
95      public void tyhjaRuudukko() {
96          for (int i = 0; i < korkeus; i++) {
97              for (int j = 0; j < leveys; j++) {
98                  StackPane stack;
99                  Ruutu ruutu = ruudukko.getRuudukko()[j][i];
100                 if (ruutu.isLiputettu()) {
101                     stack = new StackPane(new ImageView(lippu));
102                 } else {
103                     stack = new StackPane(new ImageView(klikkaamaton));
104                 }
105                 int leveysNyt = j;
106                 int korkeusNyt = i;
107                 stack.setOnMouseClicked(e -> {
108                     if (e.getButton().equals(MouseButton.PRIMARY)) {
109                         if (!ruutu.isLiputettu()) {
110                             ruudukko.kokoaRuudukko(leveysNyt, korkeusNyt);
111                             paivita(miinaKentta);
112                         }
113                     } else if (e.getButton().equals(MouseButton.SECONDARY)) {
114                         ruudukko.liputus(leveysNyt, korkeusNyt);
115                         tyhjaRuudukko();
116                     }
117                 });
118                 miinaKentta.add(stack, j, i);
119             }
120         }
121     }
122 
123     /**
124      * Päivittää näkymän pelitilanteen mukaan.
125      *
126      * @param grid miinakenttä visualisoituna
127      */
128     public void paivita(GridPane grid) {
129         if (ruudukko.getTila().equals(Ruudukkotila.VOITTOTILA)) {
130             timeline.stop();
131             voitto();
132             return;
133         }
134         for (int i = 0; i < ruudukko.getKorkeus(); i++) {
135             for (int j = 0; j < ruudukko.getLeveys(); j++) {
136                 Ruutu ruutu = ruudukko.getRuudukko()[j][i];
137                 StackPane stack;
138                 int leveysNyt = j;
139                 int korkeusNyt = i;
140                 if (ruutu.isKlikattu()) {
141                     if (ruutu.getTyyppi().equals(Ruututyyppi.MIINA)) {
142                         havio(leveysNyt, korkeusNyt);
143                         return;
144                     } else {
145                         stack = new StackPane(new ImageView(new Image("tyhja" + ruutu.getYmparys() + ".png")));
146                     }
147                 } else if (ruutu.isLiputettu()) {
148                     stack = new StackPane(new ImageView(lippu));
149                     stack.setOnMouseClicked(e -> {
150                         if (e.getButton().equals(MouseButton.SECONDARY)) {
151                             ruudukko.liputus(leveysNyt, korkeusNyt);
152                             paivita(grid);
153                         }
154                     });
155                 } else {
156                     stack = new StackPane(new ImageView(klikkaamaton));
157                     stack.setOnMouseClicked(e -> {
158                         if (e.getButton().equals(MouseButton.SECONDARY)) {
159                             ruudukko.liputus(leveysNyt, korkeusNyt);
160                             paivita(grid);
161                         } else if (e.getButton().equals(MouseButton.PRIMARY)) {
162                             ruudukko.klikkaus(leveysNyt, korkeusNyt);
163                             paivita(grid);
164                         }
165                     });
166                 }
167                 grid.add(stack, j, i);
168             }
169         }
170     }

172     /**
173      * Hoitaa tilanteen, jossa pelaaja on voittanut pelin.
174      */
175     public void voitto() {
176         int aika = Integer.parseInt(aikaLabel.getText());
177         VoittoNakyma nakyma = new VoittoNakyma(ruudukko, stage, aika);
178         stage.setScene(nakyma.scene());
179     }
180 
181     /**
182      * Hoitaa tilanteen, jossa pelaaja on hävinnyt pelin.
183      * @param vikaLeveys viimeisen klikkauksen x-koordinaatti
184      * @param vikaKorkeus viimeisen klikkauksen y-koordinaatti
185      */
186     public void havio(int vikaLeveys, int vikaKorkeus) {
187         int aika = Integer.parseInt(aikaLabel.getText());
188         HavioNakyma nakyma = new HavioNakyma(ruudukko, stage, aika, vikaLeveys, vikaKorkeus);
189         stage.setScene(nakyma.scene());
190     }
