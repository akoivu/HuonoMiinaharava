6   package huonomiinaharava.kayttoliittyma;
7   
8   import huonomiinaharava.huonomiinaharava.logiikka.Ruudukko;
9   import huonomiinaharava.huonomiinaharava.logiikka.Ruutu;
10  import javafx.scene.Scene;
11  import javafx.scene.control.Label;
12  import javafx.scene.image.Image;
13  import javafx.scene.image.ImageView;
14  import javafx.scene.layout.GridPane;
15  import javafx.scene.layout.StackPane;
16  import javafx.scene.layout.VBox;
17  import javafx.stage.Stage;
18  
19  /**
20   * Näkymä, joka kootaan pelaajan voittaessa pelin.
21   * @author anttikoivurova
22   */
23  public class VoittoNakyma {
24  
25      private VBox kokonaisuus;
26      private GridPane grid;
27      private Ruudukko ruudukko;
28      private Stage stage;
29      private int aika;
30      private Image lippu = new Image("lippu.png");
31      private Image nolla = new Image("tyhja0.png");
32      private Image ykkonen = new Image("tyhja1.png");
33      private Image kakkonen = new Image("tyhja2.png");
34      private Image kolmonen = new Image("tyhja3.png");
35      private Image nelonen = new Image("tyhja4.png");
36      private Image vitonen = new Image("tyhja5.png");
37      private Image kutonen = new Image("tyhja6.png");
38      private Image seiska = new Image("tyhja7.png");
39      private Image kasi = new Image("tyhja8.png");
40  
41      /**
42       * Luo uuden voittonäkymän.
43       * @param ruudukko pelissä käytetty ruudukko
44       * @param stage primaryStage
45       * @param aika pelin päätösaika
46       */
47      public VoittoNakyma(Ruudukko ruudukko, Stage stage, int aika) {
48          this.ruudukko = ruudukko;
49          this.stage = stage;
50          this.aika = aika;
51  
52          kokoaNakyma();
53      }
54  
55      /**
56       * Palauttaa näkymän.
57       * @return näkymä
58       */
59      public Scene scene() {
60          return new Scene(kokonaisuus);
61      }
62  
63      /**
64       * Kokoaa voittonäkymän.
65       */
66      public void kokoaNakyma() {
67          grid = new GridPane();
68          kokonaisuus = new VBox(10);
69          Label voittoteksti = new Label("Voi juku, voitit! Aikasi oli " + aika + " sekuntia.");
70          kokonaisuus.getChildren().add(voittoteksti);
71          for (int i = 0; i < ruudukko.getKorkeus(); i++) {
72              for (int j = 0; j < ruudukko.getLeveys(); j++) {
73                  Ruutu ruutu = ruudukko.getRuudukko()[j][i];
74                  StackPane stack;
75                  if (ruutu.isKlikattu()) {
76                      stack = new StackPane(new ImageView(new Image("tyhja" + ruutu.getYmparys() + ".png")));
77                  } else {
78                      stack = new StackPane(new ImageView(lippu));
79                  }
80                  grid.add(stack, j, i);
81              }
82          }
83  
84          kokonaisuus.getChildren().add(grid);
85      }
86  }
